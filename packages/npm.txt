# npm.txt - global npm packages
# Usage: install/node.sh reads this file
# Install manually: npm install -g <package>
#
# Entries may pin a version: <name>@<version>. Pinned packages are held at
# that exact version (upgrade mode does not move them) — bump the pin here
# deliberately.

# pi (https://pi.dev) — multi-provider coding agent. Official installer at
# pi.dev/install.sh ultimately runs `npm install -g @earendil-works/pi-coding-agent`,
# so we use that directly to keep this list authoritative.
@earendil-works/pi-coding-agent

# OpenAI Codex CLI — the npm tarball is just a thin wrapper around the same
# Rust binary published on github.com/openai/codex/releases. We use npm so
# upgrades happen via this list, not a hand-rolled GitHub release fetcher.
# install/codex.sh still handles managed ~/.codex config + hooks + healthcheck.
#
# UNPINNED: `bootstrap upgrade` always pulls latest. codex.sh runs right after
# node.sh and reconciles the managed ~/.codex config (profiles are per-file
# since 0.134), so a config-format shift surfaces in that step's healthcheck
# rather than silently. If codex ever breaks config compatibility again,
# re-pin here (`@openai/codex@X.Y.Z`) until the config catches up.
@openai/codex

# qmd — local hybrid search (BM25+vectors+rerank, all-local GGUF models) over
# markdown knowledge: ~/kb + auto-memory + dotfiles docs. Serves one warm MCP
# daemon on :8181 for Claude+Codex+opencode. See install/memory.sh.
@tobilu/qmd

# Playwright CLI — token-efficient browser automation (artifacts to disk,
# agent reads what it needs) vs streaming MCP trees. chrome-devtools MCP
# stays for live inspection; this is the automation path (~4x fewer tokens,
# attributed to the Playwright team's launch materials).
@playwright/cli

# LSP servers backing the Claude Code LSP plugins (pyright-lsp, typescript-lsp).
# The plugins only wrap these binaries — without them the plugins are inert.
# (rust-analyzer comes from a rustup component in install/rust.sh; clangd from
# Xcode on macOS / brew llvm on Linux.)
pyright
typescript-language-server
typescript
