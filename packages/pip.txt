# pip.txt - Python CLI tools installed via `uv tool install`
# Usage: install/python.sh reads this file
#
# Each tool gets its own isolated venv — no conflicts, no monolithic environment.
# Libraries (numpy, scipy, etc.) belong in per-project pyproject.toml, not here.

# Interactive Python
ipython

# Jupyter
jupyterlab

# Monitoring
glances

# CLI/output helpers
rich-cli

# Local LLM inference
mlx-lm  # macos-only (Apple MLX framework requires Metal GPU)
mlx-openai-server  # macos-only python=3.12 (outlines-core 0.1.26 has no cp313/cp314 wheels; pinned via outlines<1.2)

# C/C++ package manager
conan

# Manim Community Edition — programmatic animation engine for explainer/education
# videos and math/sim visualizations. Needs cairo/pango/pkg-config (Brewfile) +
# ffmpeg (already present); LaTeX is optional (only Tex/MathTex use it).
manim

# MkDocs — static documentation-site generator from Markdown + a YAML config.
# `mkdocs build --strict` surfaces broken links/config as CI failures.
mkdocs
