# cargo.txt - Rust tools to install via cargo-binstall
# Usage: install/rust.sh reads this file
#
# Only include tools that have pre-built binaries on GitHub Releases.
# cargo-binstall checks for a release binary first, falling back to source
# compilation — but source compilation fails under macOS Sequoia's sandbox.
# Tools with Homebrew bottles go in packages/Brewfile instead.

# Better file finder (pre-built: github.com/sharkdp/fd)
fd-find

# Better sed (pre-built: github.com/chmln/sd)
sd

# Benchmarking (pre-built: github.com/sharkdp/hyperfine)
hyperfine

# Better rm/rip (pre-built: github.com/nivekuil/rip)
rm-improved

# Smarter cd (pre-built: github.com/ajeetdsouza/zoxide)
zoxide

# Cat with syntax highlighting (pre-built: github.com/sharkdp/bat)
bat

# Better grep (pre-built: github.com/BurntSushi/ripgrep)
ripgrep

# Git diff viewer (pre-built: github.com/dandavison/delta)
git-delta

# Structural diff (pre-built: github.com/Wilfred/difftastic)
difftastic

# Better ps (pre-built: github.com/dalance/procs)
procs

# System monitor (pre-built: github.com/ClementTsang/bottom)
bottom

# Structural code search (pre-built: github.com/ast-grep/ast-grep)
ast-grep

# Modern shell (pre-built: github.com/nushell/nushell)
nu

# CSV toolkit (pre-built: github.com/medialab/xan)
xan

# Command runner / Makefile alternative (pre-built: github.com/casey/just)
just

# Markup-based typesetting (LaTeX alternative, pre-built: github.com/typst/typst)
typst-cli

# mdBook diagram preprocessor (pre-built: github.com/badboy/mdbook-mermaid)
# mdbook itself comes from the Brewfile (Homebrew has it; this plugin doesn't).
mdbook-mermaid

# Shell history: SQLite-backed search + optional e2e-encrypted sync (pre-built: github.com/atuinsh/atuin)
# NOTE: ships aarch64-apple-darwin but no x86_64-apple-darwin asset — Intel Macs
# would need the Brewfile instead (none in the fleet today).
atuin

# Rust test runner: parallel, per-test process isolation (pre-built: github.com/nextest-rs/nextest)
cargo-nextest

# Unit-aware scientific calculator (pre-built: github.com/sharkdp/numbat)
# Crate is numbat-cli; the bare `numbat` crate is the library and has no binary.
numbat-cli

# Hex viewer with color-coded byte categories (pre-built: github.com/sharkdp/hexyl)
hexyl

# Sampling CPU profiler → Firefox Profiler UI, no sudo needed (pre-built: github.com/mstange/samply)
samply

# httpie-style HTTP client (pre-built: github.com/ducaale/xh)
xh

# tldr client: example-first command docs, offline cache (pre-built: github.com/tealdeer-rs/tealdeer)
# Binary is `tldr`.
tealdeer

# Visual disk usage tree (pre-built: github.com/bootandy/dust). Crate is du-dust; binary `dust`.
du-dust

# Source-aware spell checker, ultra-low false positives (pre-built: github.com/crate-ci/typos)
# Crate is typos-cli; binary `typos`.
typos-cli

# Single-binary static file server with upload/QR (pre-built: github.com/svenstaro/miniserve)
miniserve

# Gitignore-aware file watcher with process-group restarts (pre-built: github.com/watchexec/watchexec)
# Crate is watchexec-cli (bare `watchexec` is the library); generic complement to entr/bacon.
watchexec-cli

# Async terminal file manager with previews; uses fd/rg/fzf/zoxide as backends
# (pre-built via binstall metadata: github.com/sxyazi/yazi)
yazi-fm
yazi-cli

# Jujutsu: git-compatible VCS — working copy is always a commit, first-class
# conflicts (pre-built: github.com/jj-vcs/jj). Crate jj-cli; binary `jj`.
# Trial: coexists with git in the same repo; not a git replacement yet.
jj-cli
